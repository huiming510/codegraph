# 插件重构详细设计：规则文件驱动的跨文件分析引擎

## 概述

本文档是 [2026-06-04-插件重构-需求.MD](./2026-06-04-插件重构-需求.MD) 的实现级设计，定义从零构建规则引擎的完整技术方案。文档按 **FEATURE 1~8** 分阶段推进，每阶段为独立可交付单元，阶段间有明确依赖。

### 设计原则

1. **不破坏现有架构**：新规则引擎与 Python 插件双轨并行，共存于 `stage6`
2. **复用现有原子能力**：复用 `stage1~5` 的 Tree-sitter 解析、符号表、类型推断结果
3. **规则即数据**：YAML 规则文件经解析后直接驱动执行，无中间代码生成
4. **渐进迁移**：每迁移一个 Python 插件为一条规则，同步验证结果一致性

### 术语约定

| 术语 | 定义 |
|------|------|
| **规则文件** | 描述单一跨文件关系的 YAML 文件（`.yaml`） |
| **规则包** | 包含 `metadata.yaml` + 多个规则文件的目录 |
| **规则引擎** | 执行规则文件加载→解析→匹配→边生成全流程的 Python 模块 |
| **Rule AST** | 规则文件解析后的内存结构（ dataclass 树） |
| **原子查询** | 直接对 AST/符号表/图谱数据的查询操作 |
| **组合查询** | 由多个原子查询通过 AND/OR/NOT 组合而成的查询 |

```python
# FEATURE1 产出：scripts/plugins/rules_engine/schema.py
from __future__ import annotations

from pydantic import BaseModel, Field, field_validator
from typing import Literal, Optional
from enum import Enum


class RuleVersion(str, Enum):
    V1 = "1"


class MatchTarget(BaseModel):
    """匹配目标：描述被匹配文件/符号的过滤条件。"""
    language: Optional[str] = None              # e.g. "java", "xml"
    file_pattern: Optional[str] = None         # glob pattern, e.g. "**/*Mapper.xml"
    annotation: Optional[str] = None            # e.g. "@Mapper"
    class_annotation: Optional[str] = None     # e.g. "@Entity"
    method_pattern: Optional[str] = None       # regex, e.g. "findBy|get|load"
    return_type: Optional[str] = None           # e.g. "List<*>"
    kind: Optional[str] = None                 # symbol kind: class, method, field, etc.
    # 跨文件条件：另一个匹配目标的约束
    paired_with: Optional[MatchTarget] = None  # 对端文件需满足的条件


class ConditionType(str, Enum):
    """条件类型：决定如何对匹配目标执行查询。"""
    AST_QUERY = "ast_query"          # Tree-sitter query（原子）
    XPATH = "xpath"                   # XPath 查询（XML）
    REGEX = "regex"                   # 正则表达式匹配
    ANNOTATION = "annotation"         # 注解匹配
    NAMING = "naming"                 # 命名模式匹配
    RELATION = "relation"             # 符号间关系（跨文件）


class BindingSpec(BaseModel):
    """绑定规格：从 AST 查询中提取命名capture的结果。"""
    name: str                         # capture 名称, e.g. "property"
    source: Literal["attr", "text", "node"] = "attr"
    transform: Optional[str] = None    # 可选变换: "camel_to_snake", "snake_to_camel", "trim"


class Condition(BaseModel):
    """单一匹配条件：描述对匹配目标执行什么查询。"""
    type: ConditionType
    # AST_QUERY / REGEX 时：查询字符串
    query: Optional[str] = None
    # ANNOTATION 时：注解名（精确或模式）
    annotation: Optional[str] = None
    # NAMING 时：正则模式
    pattern: Optional[str] = None
    # RELATION 时：关联目标的 key
    relates_to: Optional[str] = None
    # 绑定提取：capture 名 → 属性路径
    bindings: dict[str, str] = Field(default_factory=dict)
    # 组合逻辑
    combinator: Optional[Literal["and", "or", "not"]] = None
    sub_conditions: list[Condition] = Field(default_factory=list)

    @field_validator("combinator", mode="before")
    def _validate_combinator(cls, v):
        if v is not None and v not in ("and", "or", "not"):
            raise ValueError(f"Invalid combinator: {v}")
        return v


class EmitEdge(BaseModel):
    """边发射规格：从匹配结果生成什么样的边。"""
    name: str = Field(description="边的标识名（用于日志/debug）")
    from_ref: str = Field(
        description="""
        边起点引用。支持三种形式：
        - "{binding_name}"         → 使用绑定变量（来自 conditions.bindings）
        - "target:{xpath}"         → 从 target 文件的 AST 中取值
        - "symbol:{qname}"          → 使用符号表中的 qname
        - "meta:source" / "meta:target" → 使用 match 中的 source/target 元信息
        """
    )
    to_ref: str = Field(description="边终点引用，同上")
    kind: str = Field(description="边类型，如 'sql-mapping', 'renders', 'form-bound'")
    properties: dict[str, str] = Field(default_factory=dict)
    line_ref: Optional[str] = None  # "{binding_name}" → 从哪个绑定取行号


class Rule(BaseModel):
    """单一规则文件的完整模型。"""
    version: RuleVersion = RuleVersion.V1

    # ── 元信息 ──────────────────────────────────────────────────────────────
    id: str = Field(description="规则唯一标识，snake_case，如 'mybatis-orm'")
    name: dict[str, str] = Field(
        default_factory=dict,
        description="多语言名称: {zh: ..., en: ...}"
    )
    description: Optional[str] = None
    version: str = Field(default="1.0.0")
    author: Optional[str] = None
    tags: list[str] = Field(default_factory=list)

    # ── 匹配范围 ────────────────────────────────────────────────────────────
    match: dict[str, MatchTarget] = Field(
        description="定义 source 和 target 两类匹配目标。"
        "source 通常是配置文件/注解（如 Mapper.xml、@Mapper）；"
        "target 通常是源代码（如 @Entity 类）。"
    )

    # ── 匹配条件 ────────────────────────────────────────────────────────────
    conditions: list[Condition] = Field(default_factory=list)

    # ── 边发射 ──────────────────────────────────────────────────────────────
    edges: list[EmitEdge] = Field(default_factory=list)

    # ── 执行控制 ────────────────────────────────────────────────────────────
    priority: int = Field(default=100, description="执行优先级，数字越小越先执行")
    enabled: bool = Field(default=True)
    # 依赖其他规则的输出（高级特性，FEATURE5 引入）
    depends_on: list[str] = Field(default_factory=list)


class RuleBundle(BaseModel):
    """规则包元数据。"""
    id: str
    name: dict[str, str]
    version: str
    description: Optional[str] = None
    author: Optional[str] = None
    license: str = "MIT"
    homepage: Optional[str] = None
    dependencies: list[dict] = Field(default_factory=list)
    targets: dict[str, list[str]] = Field(
        default_factory=lambda: {"languages": [], "frameworks": []}
    )
    rules: list[dict] = Field(default_factory=list)
```

```python
# FEATURE1 产出: scripts/plugins/rules_engine/schema.py

from pydantic import ValidationError
from typing import Union

def validate_rule(yaml_content: str | dict) -> tuple[Rule, list[str]]:
    """
    校验规则文件，返回 (Rule 对象, 错误列表)。
    错误为空列表表示校验通过。
    """
    errors = []
    try:
        if isinstance(yaml_content, str):
            import yaml as _yaml
            data = _yaml.safe_load(yaml_content)
        else:
            data = yaml_content

        rule = Rule.model_validate(data)
        return rule, []
    except ValidationError as e:
        for err in e.errors():
            field = ".".join(str(l) for l in err["loc"])
            errors.append(f"  [{field}] {err['msg']}: {err['input']}")
        return None, errors
    except Exception as e:
        return None, [f"  Parse error: {e}"]


def validate_bundle(bundle_dir: Path) -> tuple[bool, list[str]]:
    """校验规则包的完整性。"""
    errors = []
    metadata_file = bundle_dir / "metadata.yaml"
    if not metadata_file.exists():
        return False, ["  metadata.yaml not found"]

    import yaml
    metadata = yaml.safe_load(metadata_file.read_text(encoding="utf-8"))
    bundle = RuleBundle.model_validate(metadata)

    rules_dir = bundle_dir / "rules"
    if not rules_dir.exists():
        return False, ["  rules/ directory not found"]

    for rule_entry in bundle.rules:
        rule_file = rules_dir / rule_entry["file"]
        if not rule_file.exists():
            errors.append(f"  Rule file not found: {rule_file}")
            continue
        _, rule_errors = validate_rule(rule_file.read_text(encoding="utf-8"))
        errors.extend(rule_errors)

    return len(errors) == 0, errors
```

## FEATURE 2：规则引擎核心（加载器、解析器、校验器）

### 2.1 核心数据流

```
YAML 文件
    │
    ▼
RuleLoader.load(bundle_path)
    │  ① 发现 bundle 目录
    │  ② 读取 metadata.yaml
    │  ③ 验证 bundle 完整性
    │
    ▼
RuleParser.parse(file_content)
    │  ① YAML → dict
    │  ② Pydantic 验证（Schema）
    │  ③ dict → Rule AST
    │
    ▼
RuleValidator.validate(rule_ast)
    │  ① 引用完整性（depends_on 是否存在）
    │  ② 循环依赖检测
    │  ③ binding 变量名合法性
    │  ④ AST query 语法预检
    │
    ▼
CompiledRule
    │  预处理的运行时结构：
    │  - Tree-sitter query 预编译
    │  - 文件 glob 索引
    │  - 符号过滤函数
    │
    ▼
RuleEngine.execute(compiled_rules, context)
```

```python
# FEATURE2 产出: scripts/plugins/rules_engine/loader.py

from pathlib import Path
from typing import Iterator
import importlib.resources

class RuleLoader:
    """
    负责从多个来源加载规则包。

    加载顺序（后面的优先）：
    1. 内置规则包（安装路径下的 built-in/）
    2. 用户本地规则包（项目根目录 .codegraph/rules/）
    3. 已安装的规则包（通过 CLI 安装的）
    4. 规则包覆盖配置（rules.yaml 中的 overrides）
    """

    def __init__(
        self,
        project_root: Path | None = None,
        cache_dir: Path | None = None,
    ):
        self.project_root = project_root or Path.cwd()
        self.cache_dir = cache_dir or self._default_cache_dir()

    def load_builtin(self) -> list[CompiledRuleBundle]:
        """加载内置规则包。"""
        bundles = []
        # 方式1：从安装包的 rules_engine/bundles/built-in/ 加载
        try:
            ref = importlib.resources.files("plugins.rules_engine.bundles.built_in")
            for entry in ref.iterdir():
                if entry.is_dir():
                    bundle_path = Path(entry.path)
                    bundle = self._load_bundle(bundle_path)
                    if bundle:
                        bundles.append(bundle)
        except (ModuleNotFoundError, TypeError):
            pass

        # 方式2：从 scripts/plugins/rules_engine/bundles/ 加载（开发模式）
        dev_path = Path(__file__).parent.parent / "bundles" / "built-in"
        if dev_path.exists():
            for subdir in dev_path.iterdir():
                if subdir.is_dir():
                    bundle = self._load_bundle(subdir)
                    if bundle:
                        bundles.append(bundle)

        return bundles

    def load_project_rules(self) -> list[CompiledRuleBundle]:
        """加载项目本地的 .codegraph/rules/ 规则包。"""
        project_rules = self.project_root / ".codegraph" / "rules"
        if not project_rules.exists():
            return []

        # 本地规则视为单一匿名包
        bundle = self._load_bundle(project_rules, name="project-local")
        return [bundle] if bundle else []

    def load_installed(self) -> list[CompiledRuleBundle]:
        """加载通过 CLI 安装的规则包。"""
        installed_dir = self.cache_dir / "installed"
        if not installed_dir.exists():
            return []

        bundles = []
        for subdir in installed_dir.iterdir():
            if subdir.is_dir():
                bundle = self._load_bundle(subdir)
                if bundle:
                    bundles.append(bundle)
        return bundles

    def _load_bundle(self, bundle_path: Path, name: str | None = None) -> CompiledRuleBundle | None:
        """加载单个规则包。"""
        from .schema import validate_bundle, RuleBundle
        import yaml

        metadata_file = bundle_path / "metadata.yaml"
        if not metadata_file.exists():
            return None

        valid, errors = validate_bundle(bundle_path)
        if not valid:
            import warnings
            warnings.warn(f"Rule bundle {bundle_path.name} has validation errors:\n" + "\n".join(errors))
            return None

        metadata = yaml.safe_load(metadata_file.read_text(encoding="utf-8"))
        bundle = RuleBundle.model_validate(metadata)

        rules_dir = bundle_path / "rules"
        compiled_rules = []
        for rule_entry in bundle.rules:
            rule_file = rules_dir / rule_entry["file"]
            if not rule_file.exists():
                continue
            compiled = self._compile_rule(rule_file, bundle_path)
            if compiled:
                compiled_rules.append(compiled)

        return CompiledRuleBundle(
            metadata=bundle,
            rules=compiled_rules,
            path=bundle_path,
        )

    def _compile_rule(self, rule_file: Path, bundle_path: Path) -> CompiledRule | None:
        """将规则文件编译为运行时结构。"""
        from .schema import validate_rule, Rule
        from .v1.rule_ast import RuleASTBuilder

        content = rule_file.read_text(encoding="utf-8")
        rule_obj, errors = validate_rule(content)
        if errors:
            import warnings
            warnings.warn(f"Rule {rule_file.name} validation errors:\n" + "\n".join(errors))
            return None

        # 构建运行时 AST
        ast_builder = RuleASTBuilder(rule_obj)
        ast = ast_builder.build()

        # 预编译 Tree-sitter 查询
        ast = self._precompile_queries(ast)

        return CompiledRule(
            rule=rule_obj,
            ast=ast,
            source_file=rule_file,
            bundle_path=bundle_path,
        )

    def _precompile_queries(self, ast) -> RuleAST:
        """预编译 AST 中的 Tree-sitter 查询。"""
        from tree_sitter import Parser
        from .v1.operators import QUERY_LANGUAGE_MAP

        for condition in ast.walk_conditions():
            if condition.type == ConditionType.AST_QUERY and condition.query:
                lang = condition.language or "java"
                lang_obj = QUERY_LANGUAGE_MAP.get(lang)
                if lang_obj:
                    try:
                        condition._compiled_query = lang_obj.query(condition.query)
                    except Exception as e:
                        condition._query_error = str(e)
        return ast
```

```python
from dataclasses import dataclass, field
from typing import Optional

from ..schema import (
    Rule, Condition, ConditionType, EmitEdge,
    MatchTarget, BindingSpec
)


@dataclass
class CompiledCondition:
    """条件的运行时表示。"""
    type: ConditionType
    query: Optional[str] = None
    annotation: Optional[str] = None
    pattern: Optional[str] = None
    bindings: dict[str, str] = field(default_factory=dict)

    # 运行时数据（由 RuleASTBuilder 填充）
    _compiled_query = None        # 预编译的 tree-sitter Query 对象
    _query_error: str | None = None
    language: str = "java"


@dataclass
class CompiledEmitEdge:
    """边发射规格的运行时表示。"""
    name: str
    from_ref: str          # 模板字符串，如 "target:{class}.{prop}"
    to_ref: str
    kind: str
    properties: dict[str, str] = field(default_factory=dict)
    line_ref: Optional[str] = None

    # 预解析的引用
    _from_parts: list = field(default_factory=list)
    _to_parts: list = field(default_factory=list)


@dataclass
class CompiledMatchTarget:
    """匹配目标的运行时表示。"""
    language: Optional[str] = None
    file_patterns: list[str] = field(default_factory=list)
    annotation: Optional[str] = None
    class_annotation: Optional[str] = None
    method_pattern: Optional[str] = None
    return_type: Optional[str] = None
    kind: Optional[str] = None
    paired_with: Optional[CompiledMatchTarget] = None

    # 预编译的 glob matcher
    _matcher = None


@dataclass
class RuleAST:
    """
    规则的完整抽象语法树。
    从 Pydantic Rule 模型转换而来，供规则引擎执行使用。
    """
    rule: Rule
    id: str
    version: str

    source: CompiledMatchTarget
    target: CompiledMatchTarget

    conditions: list[CompiledCondition] = field(default_factory=list)
    edges: list[CompiledEmitEdge] = field(default_factory=list)

    priority: int = 100
    enabled: bool = True
    depends_on: list[str] = field(default_factory=list)

    def walk_conditions(self):
        """遍历所有条件节点。"""
        for c in self.conditions:
            yield c
```

#### 2.2.3 RuleASTBuilder

```python
# FEATURE2 产出: scripts/plugins/rules_engine/v1/rule_ast.py

class RuleASTBuilder:
    """
    将 Pydantic Rule 模型转换为运行时 RuleAST。
    负责：默认值填充、默认值推导、跨字段一致性检查。
    """

    def __init__(self, rule: Rule):
        self.rule = rule

    def build(self) -> RuleAST:
        rule = self.rule

        # 推断 language（从第一个 condition 或 match 中）
        inferred_lang = self._infer_language()

        # 构建 source/target
        source = self._build_match_target(
            rule.match.get("source", {}), inferred_lang
        )
        target = self._build_match_target(
            rule.match.get("target", {}), inferred_lang
        )

        # 构建条件列表
        conditions = self._build_conditions(rule.conditions, inferred_lang)

        # 构建边发射规格
        edges = self._build_edges(rule.edges)

        return RuleAST(
            rule=rule,
            id=rule.id,
            version=rule.version,
            source=source,
            target=target,
            conditions=conditions,
            edges=edges,
            priority=rule.priority,
            enabled=rule.enabled,
            depends_on=rule.depends_on,
        )

    def _infer_language(self) -> str:
        """从 match 条件中推断语言。"""
        match = self.rule.match
        for key in ("source", "target"):
            mt = match.get(key)
            if mt and mt.language:
                return mt.language
        return "java"  # 默认

    def _build_match_target(self, mt: MatchTarget, default_lang: str) -> CompiledMatchTarget:
        patterns = []
        if mt.file_pattern:
            patterns.append(mt.file_pattern)
        return CompiledMatchTarget(
            language=mt.language or default_lang,
            file_patterns=patterns,
            annotation=mt.annotation,
            class_annotation=mt.class_annotation,
            method_pattern=mt.method_pattern,
            return_type=mt.return_type,
            kind=mt.kind,
            paired_with=self._build_match_target(mt.paired_with, default_lang)
                           if mt.paired_with else None,
        )

    def _build_conditions(self, conditions: list[Condition], default_lang: str) -> list[CompiledCondition]:
        compiled = []
        for cond in conditions:
            compiled_cond = CompiledCondition(
                type=cond.type,
                query=cond.query,
                annotation=cond.annotation,
                pattern=cond.pattern,
                bindings=dict(cond.bindings),
                language=default_lang,
            )
            if cond.sub_conditions:
                compiled_cond._sub = self._build_conditions(cond.sub_conditions, default_lang)
            compiled.append(compiled_cond)
        return compiled

    def _build_edges(self, edges: list[EmitEdge]) -> list[CompiledEmitEdge]:
        compiled = []
        for edge in edges:
            from_parts = self._parse_ref_template(edge.from_ref)
            to_parts = self._parse_ref_template(edge.to_ref)
            compiled_edge = CompiledEmitEdge(
                name=edge.name,
                from_ref=edge.from_ref,
                to_ref=edge.to_ref,
                kind=edge.kind,
                properties=dict(edge.properties),
                line_ref=edge.line_ref,
                _from_parts=from_parts,
                _to_parts=to_parts,
            )
            compiled.append(compiled_edge)
        return compiled

    def _parse_ref_template(self, template: str) -> list[dict]:
        """
        解析引用模板，提取绑定变量和字面量片段。
        例如: "target:{class}.{prop}" → [
            {"type": "literal", "value": "target:"},
            {"type": "binding", "name": "class"},
            {"type": "literal", "value": "."},
            {"type": "binding", "name": "prop"},
        ]
        """
        import re
        parts = []
        last_end = 0
        for m in re.finditer(r"\{([^}]+)\}", template):
            if m.start() > last_end:
                parts.append({"type": "literal", "value": template[last_end:m.start()]})
            parts.append({"type": "binding", "name": m.group(1)})
            last_end = m.end()
        if last_end < len(template):
            parts.append({"type": "literal", "value": template[last_end:]})
        return parts
```

```python
# FEATURE3 产出: scripts/plugins/rules_engine/matcher.py

import fnmatch
import re
from typing import Iterator


class MatchFilter:
    """
    快速预过滤器：根据 match.source / match.target 的表面条件，
    将待处理范围快速缩小到候选文件集。

    不做深度 AST 分析，只做：
    - glob 文件名匹配
    - 注解存在性检查（通过符号表）
    - 符号 kind 过滤
    """

    def __init__(self, ast: RuleAST, context: GraphContext):
        self.ast = ast
        self.context = context

    def filter_candidates(self) -> list[MatchPair]:
        """
        返回所有满足 match.source 和 match.target 条件的文件对。
        每个 MatchPair 包含 source_file, target_file, 以及预过滤信息。
        """
        source_candidates = self._find_files(self.ast.source)
        target_candidates = self._find_files(self.ast.target)

        pairs = []
        # 全遍历模式：source × target（规则引擎默认模式）
        for src in source_candidates:
            for tgt in target_candidates:
                if src["file"] == tgt["file"]:
                    continue  # 跳过同文件（单文件规则单独处理）
                pairs.append(MatchPair(
                    source_file=src["file"],
                    target_file=tgt["file"],
                    source_symbols=src.get("symbols", []),
                    target_symbols=tgt.get("symbols", []),
                ))
        return pairs

    def _find_files(self, target: CompiledMatchTarget) -> list[dict]:
        """根据 match target 条件找到所有匹配文件。"""
        results = []
        lang = target.language or "java"

        for sym in self.context.symbols:
            if sym.get("lang") != lang and sym.get("lang") != "":
                continue

            fp = sym.get("file", "")
            if not fp:
                continue

            # 文件名 glob 过滤
            if target.file_patterns:
                if not any(fnmatch.fnmatch(fp, pat) for pat in target.file_patterns):
                    continue

            # 注解过滤
            if target.class_annotation:
                if target.class_annotation not in (sym.get("annotations") or []):
                    continue

            # kind 过滤
            if target.kind and sym.get("kind") != target.kind:
                continue

            # method_pattern 过滤
            if target.method_pattern:
                name = sym.get("name", "")
                if not re.match(target.method_pattern, name):
                    continue

            # 收集结果（同文件的 symbols 合并）
            if not results or results[-1]["file"] != fp:
                results.append({"file": fp, "symbols": []})
            results[-1]["symbols"].append(sym)

        return results
```

### 3.3 ConditionEvaluator（条件执行器）

```python
# FEATURE3 产出: scripts/plugins/rules_engine/matcher.py


class ConditionEvaluator:
    """
    对单个 MatchPair 执行规则中的 conditions，
    提取绑定值并返回匹配结果。
    """

    def __init__(self, ast: RuleAST, context: GraphContext):
        self.ast = ast
        self.context = context
        self.ast_data = self._build_ast_index(context)

    def evaluate(self, pair: MatchPair) -> list[BindingResult]:
        """
        对一个文件对执行所有 conditions。
        返回所有成功匹配的 BindingResult 列表。
        """
        results = []

        for cond in self.ast.conditions:
            result = self._evaluate_condition(cond, pair)
            if result:
                results.extend(result)

        return results

    def _evaluate_condition(self, cond: CompiledCondition, pair: MatchPair) -> list[BindingResult]:
        """执行单个条件。"""
        if cond.type == ConditionType.AST_QUERY:
            return self._eval_ast_query(cond, pair)
        elif cond.type == ConditionType.ANNOTATION:
            return self._eval_annotation(cond, pair)
        elif cond.type == ConditionType.NAMING:
            return self._eval_naming(cond, pair)
        elif cond.type == ConditionType.REGEX:
            return self._eval_regex(cond, pair)
        elif cond.type == ConditionType.RELATION:
            return self._eval_relation(cond, pair)
        return []

    def _eval_ast_query(self, cond: CompiledCondition, pair: MatchPair) -> list[BindingResult]:
        """执行 Tree-sitter 查询并提取绑定。"""
        if cond._query_error:
            return []

        # 确定查询目标文件
        file_to_query = self._file_for_condition(cond, pair)
        if not file_to_query:
            return []

        # 获取该文件的 AST
        file_ast = self._get_file_ast(file_to_query)
        if not file_ast:
            return []

        # 执行预编译的 Tree-sitter 查询
        lang = self._lang_for_file(file_to_query)
        try:
            captures = cond._compiled_query.captures(file_ast.root_node)
        except Exception:
            return []

        results = []
        for node, capture_name in captures:
            bindings = self._extract_bindings(node, cond.bindings, file_ast)
            if bindings:
                results.append(BindingResult(
                    bindings=bindings,
                    file=file_to_query,
                    line=node.start_point[0] + 1 if node.start_point else 1,
                    node=node,
                ))
        return results

    def _eval_annotation(self, cond: CompiledCondition, pair: MatchPair) -> list[BindingResult]:
        """基于注解匹配。"""
        results = []
        for sym in pair.target_symbols:
            annotations = sym.get("annotations", [])
            ann = cond.annotation
            if ann and ann in annotations:
                bindings = self._resolve_bindings_from_symbol(sym, cond.bindings)
                results.append(BindingResult(
                    bindings=bindings,
                    file=sym.get("file", ""),
                    line=sym.get("line", 1),
                    symbol=sym,
                ))
        return results

    def _eval_naming(self, cond: CompiledCondition, pair: MatchPair) -> list[BindingResult]:
        """基于命名模式匹配。"""
        import re
        results = []
        pattern = re.compile(cond.pattern or "")
        for sym in pair.target_symbols:
            name = sym.get("name", "")
            if pattern.search(name):
                bindings = self._resolve_bindings_from_symbol(sym, cond.bindings)
                results.append(BindingResult(
                    bindings=bindings,
                    file=sym.get("file", ""),
                    line=sym.get("line", 1),
                    symbol=sym,
                ))
        return results

    def _resolve_bindings_from_symbol(self, sym: dict, binding_spec: dict[str, str]) -> dict[str, str]:
        """
        从符号中解析 binding 规格。
        binding_spec: {"property": "key=property", "column": "key=column"}
        返回: {"property": "userId", "column": "user_id"}
        """
        result = {}
        for binding_name, path in binding_spec.items():
            if path.startswith("key="):
                key = path[4:]
                annotations = sym.get("annotations", [])
                # 从注解中提取 key=value
                for ann in annotations:
                    if f"{key}=" in ann:
                        m = re.search(rf'{key}="([^"]*)"', ann)
                        if m:
                            result[binding_name] = m.group(1)
            elif path == "name":
                result[binding_name] = sym.get("name", "")
            elif path == "qname":
                result[binding_name] = sym.get("qualified_name", "")
            elif path == "type_hint":
                result[binding_name] = sym.get("type_hint", "")
        return result

    def _extract_bindings(
        self, node, binding_spec: dict[str, str], file_ast
    ) -> dict[str, str]:
        """从 Tree-sitter 节点中提取绑定值。"""
        result = {}
        for binding_name, attr_path in binding_spec.items():
            value = self._get_node_value(node, attr_path, file_ast)
            if value:
                result[binding_name] = value
        return result

    def _get_node_value(self, node, path: str, file_ast) -> str:
        """
        从 Tree-sitter 节点中按路径取值。

        支持路径格式：
        - "text"          → 节点文本内容
        - "attr=X"        → 属性值（XML 属性）
        - "parent.attr=X" → 父节点的属性
        - "key=X"         → 注解参数中的 key=value 对
        """
        if path == "text":
            return node.text.decode("utf-8") if hasattr(node, "text") else ""

        if path.startswith("attr="):
            attr_name = path[5:]
            return self._get_xml_attr(node, attr_name)

        if path.startswith("key="):
            key = path[4:]
            return self._extract_annotation_kv(node, key)

        if path.startswith("parent."):
            parent_path = path[7:]
            parent = node.parent
            if parent:
                return self._get_node_value(parent, parent_path, file_ast)

        return ""

    def _build_ast_index(self, context: GraphContext) -> dict[str, dict]:
        """构建 file → AST 字典以便快速查询。"""
        index = {}
        for entry in context.ast_data:
            fp = entry.get("file", "")
            if fp:
                index[fp] = entry.get("ast", {})
        return index

    def _get_file_ast(self, file_path: str) -> dict | None:
        return self.ast_data.get(file_path)

    def _lang_for_file(self, file_path: str) -> str:
        import os
        _, ext = os.path.splitext(file_path.lower())
        return ext.lstrip(".")
```

```python
# FEATURE4 产出: scripts/plugins/rules_engine/emitter.py

from dataclasses import dataclass, field
from typing import Optional


@dataclass
class EmitContext:
    """边发射的上下文。"""
    bindings: dict[str, str]          # 绑定结果
    pair: MatchPair                  # 文件对
    rule: RuleAST                     # 规则 AST
    graph_context: GraphContext        # 图谱上下文（符号表、已有边）


class TemplateRenderer:
    """
    渲染引用模板，将 {binding} 占位符替换为实际值。
    """

    def render(self, template: str, context: EmitContext) -> str:
        """
        渲染单个引用模板。
        输入: "target:{class}.{prop}"
        输出: "target:com.example.User.id"
        """
        import re

        def replacer(m):
            var_name = m.group(1)
            if var_name == "meta:source":
                return context.pair.source_file
            if var_name == "meta:target":
                return context.pair.target_file
            if var_name in context.bindings:
                return context.bindings[var_name]
            # 尝试从符号表查找
            return self._resolve_from_symbols(var_name, context)

        return re.sub(r"\{([^}]+)\}", replacer, template)

    def _resolve_from_symbols(self, name: str, context: EmitContext) -> str:
        """尝试从符号表中解析未绑定的变量名。"""
        # 查找同名的方法/字段符号
        for sym in context.graph_context.symbols:
            if sym.get("name") == name:
                return sym.get("qualified_name", name)
        return name


class EdgeEmitter:
    """
    边发射器：根据规则中的 edges 规格，从 BindingResult 生成 Edge 对象。
    """

    def __init__(self, graph_context: GraphContext):
        self.graph_context = graph_context
        self.renderer = TemplateRenderer()
        self._seen: set[tuple] = set()  # 去重：(kind, from_qname, to_qname)

    def emit(self, results: list[BindingResult], ast: RuleAST) -> list[Edge]:
        """
        从所有匹配结果生成边列表。
        """
        edges = []
        for result in results:
            ctx = EmitContext(
                bindings=result.bindings,
                pair=MatchPair(
                    source_file=result.file,
                    target_file=result.file,
                ),
                rule=ast,
                graph_context=self.graph_context,
            )
            for emit_spec in ast.edges:
                edge = self._emit_single(emit_spec, ctx, result)
                if edge:
                    edges.append(edge)
        return edges

    def _emit_single(
        self, emit_spec: CompiledEmitEdge, ctx: EmitContext, result: BindingResult
    ) -> Edge | None:
        """生成单条边。"""
        from ..common.base import Edge

        # 渲染 from/to 引用
        from_qname = self.renderer.render(emit_spec.from_ref, ctx)
        to_qname = self.renderer.render(emit_spec.to_ref, ctx)

        if not from_qname or not to_qname:
            return None

        # 去重
        key = (emit_spec.kind, from_qname, to_qname)
        if key in self._seen:
            return None
        self._seen.add(key)

        # 渲染 properties 中的 {binding}
        properties = {}
        for k, v in emit_spec.properties.items():
            rendered = self.renderer.render(v, ctx)
            properties[k] = rendered

        # 确定行号
        line = result.line
        if emit_spec.line_ref and emit_spec.line_ref in ctx.bindings:
            line = int(ctx.bindings[emit_spec.line_ref])

        # 确定文件路径
        file_path = result.file

        return Edge(
            kind=emit_spec.kind,
            from_qname=from_qname,
            to_qname=to_qname,
            file=file_path,
            line=line,
            metadata=properties,
        )

    def classify_edge(self, edge: Edge) -> str:
        """
        根据 from_qname / to_qname 确定边的分类。
        """
        from ..constants import get_layer_from_path, get_file_type

        from_fp = self._resolve_file(edge.from_qname)
        to_fp = self._resolve_file(edge.to_qname)

        if not from_fp or not to_fp:
            return "unknown"

        from_layer = get_layer_from_path(from_fp)
        to_layer = get_layer_from_path(to_fp)
        from_type = get_file_type(from_fp)
        to_type = get_file_type(to_fp)

        if from_layer != to_layer:
            return "cross_layer"
        if from_type != to_type:
            return "cross_type"
        return "intra_type"

    def _resolve_file(self, qname: str) -> str:
        """从 qname 解析文件路径。"""
        # 1. 直接是文件路径（如 template::/path/to/file.jsp）
        if qname.startswith("template::") or qname.startswith("config::"):
            return qname.split("::", 1)[1]

        # 2. 从符号表查找
        for sym in self.graph_context.symbols:
            if sym.get("qualified_name") == qname:
                return sym.get("file", "")

        return ""
```

```python
# FEATURE5 产出: scripts/plugins/rules_engine/engine.py

class RuleEngine:
    """
    规则引擎主类。协调 RuleLoader → Matcher → Emitter 全流程。
    """

    def __init__(self, context: GraphContext, project_root: Path | None = None):
        self.context = context
        self.project_root = project_root or Path.cwd()
        self.loader = RuleLoader(project_root=self.project_root)
        self.validator = RuleValidator()
        self.filter: MatchFilter | None = None
        self.evaluator: ConditionEvaluator | None = None
        self.emitter: EdgeEmitter | None = None
        self._compiled_rules: list[CompiledRule] = []

    def load_rules(self, enabled_ids: list[str] | None = None) -> list[CompiledRule]:
        """
        加载所有可用规则，按 priority 排序。
        enabled_ids 不为空时，只加载指定的规则。
        """
        all_bundles = []
        all_bundles.extend(self.loader.load_builtin())
        all_bundles.extend(self.loader.load_project_rules())
        all_bundles.extend(self.loader.load_installed())

        compiled_rules = []
        for bundle in all_bundles:
            for rule in bundle.rules:
                if rule.rule.id in (enabled_ids or [rule.rule.id]):
                    if not rule.rule.enabled:
                        continue
                    compiled_rules.append(rule)

        # 按 priority 排序（数字小的先执行）
        compiled_rules.sort(key=lambda r: r.ast.priority)
        self._compiled_rules = compiled_rules
        return compiled_rules

    def execute(self, rules: list[CompiledRule] | None = None) -> list[Edge]:
        """
        执行规则引擎，返回生成的 Edge 列表。
        """
        rules = rules or self._compiled_rules
        self.evaluator = ConditionEvaluator(
            ast=rules[0].ast if rules else None,
            context=self.context,
        )
        self.emitter = EdgeEmitter(self.context)

        all_edges: list[Edge] = []
        for rule in rules:
            ast = rule.ast

            # 校验
            if not self.validator.validate(ast):
                import warnings
                warnings.warn(f"Rule {ast.id} validation failed:\n" +
                              "\n".join(self.validator.errors))
                continue

            # 执行
            self.filter = MatchFilter(ast, self.context)
            pairs = self.filter.filter_candidates()

            for pair in pairs:
                results = self.evaluator.evaluate(pair)
                edges = self.emitter.emit(results, ast)
                all_edges.extend(edges)

        return all_edges

    def execute_incremental(
        self, changed_files: set[str], rules: list[CompiledRule] | None = None
    ) -> list[Edge]:
        """
        增量执行：只对变更的文件相关的规则重新执行。
        规则依赖分析：找出哪些规则的 match 条件会匹配 changed_files。
        """
        rules = rules or self._compiled_rules
        affected_rules = []
        for rule in rules:
            ast = rule.ast
            for pair in self.filter.filter_candidates() if self.filter else []:
                if pair.source_file in changed_files or pair.target_file in changed_files:
                    affected_rules.append(rule)
                    break

        return self.execute(affected_rules)


def run_rule_engine(
    context: GraphContext,
    project_root: Path,
    enabled_rules: list[str] | None = None,
    verbose: bool = False,
) -> list[Edge]:
    """
    便捷入口函数。从 stage6_cross_file.py 直接调用。
    """
    engine = RuleEngine(context=context, project_root=project_root)
    engine.load_rules(enabled_ids=enabled_rules)

    if verbose:
        print(f"  Rule engine: loaded {len(engine._compiled_rules)} rules")

    edges = engine.execute()
    return edges
```

### 5.4 双轨结果对比（DiffReporter）

```python
# FEATURE5 产出: scripts/plugins/rules_engine/diff.py

class DiffReporter:
    """
    在开发模式下，对比规则引擎和 Python 插件的输出差异。
    用于验证规则迁移的正确性。
    """

    def compare(
        self,
        rule_edges: list[dict],
        plugin_edges: list[dict],
        rule_id: str,
    ) -> DiffResult:
        """
        对比两组边，报告：
        - rule_edges 有但 plugin_edges 没有的边（规则多产出）
        - rule_edges 没有但 plugin_edges 有的边（规则少产出）
        - 完全匹配的边
        """
        rule_set = self._edge_set(rule_edges)
        plugin_set = self._edge_set(plugin_edges)

        rule_only = rule_set - plugin_set
        plugin_only = plugin_set - rule_set
        common = rule_set & plugin_set

        return DiffResult(
            rule_id=rule_id,
            rule_only=list(rule_only),
            plugin_only=list(plugin_only),
            common=list(common),
            match_rate=len(common) / max(len(rule_set), 1),
        )

    def _edge_set(self, edges: list[dict]) -> set[tuple]:
        return {
            (e.get("kind"), e.get("from_qname") or e.get("from_node", ""),
             e.get("to_qname") or e.get("to_node", ""))
            for e in edges
        }

    def format_diff(self, diff: DiffResult) -> str:
        lines = [f"Rule: {diff.rule_id} | Match rate: {diff.match_rate:.1%}"]
        if diff.rule_only:
            lines.append(f"  Rule only ({len(diff.rule_only)} edges):")
            for e in diff.rule_only[:5]:
                lines.append(f"    {e[0]}: {e[1]} → {e[2]}")
            if len(diff.rule_only) > 5:
                lines.append(f"    ... and {len(diff.rule_only) - 5} more")
        if diff.plugin_only:
            lines.append(f"  Plugin only ({len(diff.plugin_only)} edges):")
            for e in diff.plugin_only[:5]:
                lines.append(f"    {e[0]}: {e[1]} → {e[2]}")
        return "\n".join(lines)
```

```python
from dataclasses import dataclass
import json


@dataclass
class InstallResult:
    success: bool
    package_id: str
    version: str = ""
    error: str = ""


class BundleManager:
    """
    规则包管理器。负责安装、卸载、查询规则包。
    规则包存储在 cache_dir / installed / {package-id}@{version} / 下。
    """

    def __init__(self, cache_dir: Path | None = None):
        self.cache_dir = cache_dir or self._default_cache_dir()
        self.installed_dir = self.cache_dir / "installed"
        self.installed_dir.mkdir(parents=True, exist_ok=True)

    def install(self, package_id: str, version: str = "latest") -> InstallResult:
        """从远程安装规则包。"""
        # 方案1：从 GitHub releases 安装
        # 方案2：从本地 tar.gz 安装
        # 方案3：从指定 URL 安装
        try:
            bundle_path = self._download_bundle(package_id, version)
            return self._extract_bundle(bundle_path, package_id, version)
        except Exception as e:
            return InstallResult(success=False, package_id=package_id, error=str(e))

    def uninstall(self, package_id: str) -> InstallResult:
        """卸载规则包。"""
        import shutil
        for subdir in self.installed_dir.iterdir():
            if subdir.is_dir() and subdir.name.startswith(package_id + "@"):
                shutil.rmtree(subdir)
                return InstallResult(success=True, package_id=package_id)
        return InstallResult(success=False, package_id=package_id,
                             error=f"Package not found: {package_id}")

    def list_installed(self) -> list[CompiledRuleBundle]:
        """列出所有已安装的规则包。"""
        from .loader import RuleLoader
        loader = RuleLoader(cache_dir=self.cache_dir)
        return loader.load_installed()

    def _download_bundle(self, package_id: str, version: str) -> Path:
        """从远程下载规则包。"""
        # TODO: 实现 GitHub releases / 规则市场的下载逻辑
        raise NotImplementedError("Remote bundle download not implemented in FEATURE6")

    def _extract_bundle(self, bundle_path: Path, package_id: str, version: str) -> InstallResult:
        """解压规则包到安装目录。"""
        import tarfile, shutil

        dest = self.installed_dir / f"{package_id}@{version}"
        dest.mkdir(parents=True, exist_ok=True)

        if bundle_path.suffix == ".tar.gz" or bundle_path.suffix == ".tgz":
            with tarfile.open(bundle_path, "r:gz") as tar:
                tar.extractall(dest)
        else:
            shutil.copytree(bundle_path, dest, dirs_exist_ok=True)

        return InstallResult(success=True, package_id=package_id, version=version)
```

```python
    from .diff import DiffReporter
    from .engine import RuleEngine
    from plugins.cross_file.plugins.mybatis_orm import MyBatisOrmPlugin
    from plugins.common.base import GraphContext

    # 构建 context
    context = build_context_from_test_repo("java1")

    # Python 插件结果
    plugin = MyBatisOrmPlugin()
    plugin_edges = plugin.produce(context)

    # 规则引擎结果
    engine = RuleEngine(context=context)
    engine.load_rules(enabled_ids=["mybatis-orm"])
    rule_edges = [e.to_dict() for e in engine.execute()]

    # 对比
    reporter = DiffReporter()
    diff = reporter.compare(rule_edges, plugin_edges, "mybatis-orm")
    print(reporter.format_diff(diff))

    # 验收条件：match_rate >= 0.90（允许 10% 的边界情况差异）
    assert diff.match_rate >= 0.90, f"Match rate too low: {diff.match_rate}"
```
